        .module('fooformsApp.core')
        .controller('HeadController', HeadController)
        .controller('MainController', MainController);

        /* jshint validthis: true */
        var vm = this;
 
        vm.activate = activate;
        vm.title = 'HeadController';
        vm.stylesheet = 'bootstrap';
 
        vm.swapStyle = function (stylesheet) {
            vm.stylesheet = stylesheet;
        };
 
        activate();
 
        ////////////////
 
        function activate() {
        }
    }
 
 
    MainController.$inject = ['$scope', 'Session'];
 
    /* @ngInject */
    function MainController($scope, Session) {
        /* jshint validthis: true */
        var vm = this;
 
        vm.activate = activate;
        vm.title = 'MainController';
 
        vm.setMessage = function (msgBox, status, title, message) {
            $scope.activeMsgBox = msgBox;
            $scope.msgStatus = status;
            $scope.msgTitle = title;
            $scope.msg = message;
        };
 
        // Allow the user to be update throughout the app using the Session service.
        $scope.$watch(function () {
            return Session.user
        }, function (newVal) {
            if (typeof newVal !== 'undefined') {
                $scope.user = Session.user;
            }
        });
        $scope.$watch(function () {
            return Session.org
        }, function (newVal) {
            if (typeof newVal !== 'undefined') {
                $scope.org = Session.org;
            }
        });
 
 
        activate();
 
        ////////////////
 
        function activate() {
            vm.sideMenuVisible = true;
 
            //Messaging throughout App
            vm.activeMsgBox = ''; // any string --matches ng-show of various msgboxes.
            vm.msgStatus = ''; // used in class, so alert-danger, etc...
            vm.msgTitle = ''; // optional -
            vm.msg = ''; // optional, but pretty stupid not to populate it
        }
 
    }
 
})();
